# Clone Repo:

- desktop (open git bash)
- git clone URL (repository)
- cd file name
- code . (VS Code)

# Front-end Config:

- cd client
- npm i / npm install
- npm run dev

# Back-end Config:

- composer install
- cp .env.example .env
- php artisan key:generate
- php artisan migrate --seed / php artisan migrate:refresh (if existing DB)
- php artisan serve
